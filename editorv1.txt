const MBtn = defineComponent({
  props: { icon: String, title: String, active: Boolean, disabled: Boolean },
  emits: ['click'],
  slots: ['icon'],
  setup(props, { emit, slots }) {
    return () => h('button', {
      title: props.title,
      disabled: props.disabled,
      class: ['mb', { 'mb--on': props.active, 'mb--off': props.disabled }],
      onMousedown: (e) => { e.preventDefault(); if (!props.disabled) emit('click') },
    }, [
      slots.icon
        ? slots.icon()
        : h('span', { class: 'material-symbols-rounded' }, props.icon)
    ])
  }
})

const HL = {
  TAG_PUNC:  '#808080',   // < > / < />
  TAG_NAME:  '#4ec9b0',   // div, span, p тАж
  ATTR_NAME: '#9cdcfe',   // class, id, href тАж
  ATTR_VAL:  '#ce9178',   // "value"
  EQUALS:    '#d4d4d4',   // =
  COMMENT:   '#6a9955',   // <!-- -->
  DOCTYPE:   '#569cd6',   // <!DOCTYPE>
  ENTITY:    '#d7ba7d',   // &amp; &#160;
  TEXT:      '#d4d4d4',   // plain text nodes
  STYLE_PROP:'#9cdcfe',   // CSS property names inside style=""
  STYLE_VAL: '#ce9178',   // CSS values inside style=""
}

function esc(s) {
  return s.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
}

function sp(color, text) {
  return `<span style="color:${color}">${text}</span>`
}

// Highlight attribute value тАУ special-case inline style=""
function hlAttrValue(raw) {
  // raw includes surrounding quotes e.g. "color:#f00;font-size:12px"
  const q = raw[0]
  if (q !== '"' && q !== "'") return sp(HL.ATTR_VAL, esc(raw))
  const inner = raw.slice(1, -1)

  // Check if it looks like a CSS declaration list (has : separators)
  if (/[\w-]+\s*:/.test(inner)) {
    // Tokenize CSS props/values
    let css = ''
    let remaining = inner
    while (remaining.length) {
      // Whitespace
      const ws = remaining.match(/^\s+/)
      if (ws) { css += esc(ws[0]); remaining = remaining.slice(ws[0].length); continue }
      // Property: value; pattern
      const decl = remaining.match(/^([\w-]+)(\s*:\s*)([^;]*)(;?)/)
      if (decl) {
        css += sp(HL.STYLE_PROP, esc(decl[1]))
           + sp(HL.TAG_PUNC,    esc(decl[2]))
           + sp(HL.STYLE_VAL,   esc(decl[3]))
           + (decl[4] ? sp(HL.TAG_PUNC, ';') : '')
        remaining = remaining.slice(decl[0].length)
        continue
      }
      css += esc(remaining[0]); remaining = remaining.slice(1)
    }
    return sp(HL.ATTR_VAL, esc(q)) + css + sp(HL.ATTR_VAL, esc(q))
  }
  return sp(HL.ATTR_VAL, esc(raw))
}

// Highlight a single tag string (everything from < to >)
function highlightTag(tag) {
  let out = ''
  let i = 0

  // <
  out += sp(HL.TAG_PUNC, '&lt;')
  i = 1

  // End-tag slash
  if (tag[i] === '/') { out += sp(HL.TAG_PUNC, '/'); i++ }

  // Tag name
  const nameM = tag.slice(i).match(/^([\w-:.]+)/)
  if (nameM) { out += sp(HL.TAG_NAME, esc(nameM[1])); i += nameM[1].length }

  // Attributes
  while (i < tag.length) {
    // Trailing /> or >
    if (tag.slice(i) === '/>') { out += sp(HL.TAG_PUNC, '/&gt;'); return out }
    if (tag[i] === '>') { out += sp(HL.TAG_PUNC, '&gt;'); return out }

    // Whitespace
    const wsM = tag.slice(i).match(/^\s+/)
    if (wsM) { out += esc(wsM[0]); i += wsM[0].length; continue }

    // Attribute name
    const anM = tag.slice(i).match(/^([\w-:.@#[\]*()|^$~]+)/)
    if (!anM) { out += esc(tag[i]); i++; continue }
    out += sp(HL.ATTR_NAME, esc(anM[1]))
    i += anM[1].length

    // = sign
    if (tag[i] === '=') {
      out += sp(HL.EQUALS, '=')
      i++
      // Quoted value
      if (tag[i] === '"' || tag[i] === "'") {
        const q = tag[i]
        const end = tag.indexOf(q, i + 1)
        const endPos = end === -1 ? tag.length : end + 1
        out += hlAttrValue(tag.slice(i, endPos))
        i = endPos
      } else {
        // Unquoted value
        const uvM = tag.slice(i).match(/^([^\s>]+)/)
        if (uvM) { out += sp(HL.ATTR_VAL, esc(uvM[1])); i += uvM[1].length }
      }
    }
  }
  return out
}

// Highlight HTML entities
function hlEntities(text) {
  return esc(text).replace(
    /(&amp;[a-zA-Z]+;|&amp;#\d+;|&amp;#x[\da-fA-F]+;)/g,
    m => sp(HL.ENTITY, m)
  )
}

// Main highlighter
function syntaxHighlight(code) {
  let result = ''
  let i = 0

  while (i < code.length) {
    // HTML Comment <!-- ... -->
    if (code.startsWith('<!--', i)) {
      const end = code.indexOf('-->', i + 4)
      const endPos = end === -1 ? code.length : end + 3
      result += sp(HL.COMMENT, esc(code.slice(i, endPos)))
      i = endPos
      continue
    }

    // DOCTYPE / CDATA
    if (code.startsWith('<!', i)) {
      const end = code.indexOf('>', i)
      const endPos = end === -1 ? code.length : end + 1
      result += sp(HL.DOCTYPE, esc(code.slice(i, endPos)))
      i = endPos
      continue
    }

    // Tag
    if (code[i] === '<') {
      // Find closing > respecting quotes
      let j = i + 1
      let inQ = null
      while (j < code.length) {
        if (inQ) {
          if (code[j] === inQ) inQ = null
        } else {
          if (code[j] === '"' || code[j] === "'") inQ = code[j]
          else if (code[j] === '>') { j++; break }
        }
        j++
      }
      result += highlightTag(code.slice(i, j))
      i = j
      continue
    }

    // Plain text until next < тАФ highlight entities within
    const nextTag = code.indexOf('<', i)
    const endPos = nextTag === -1 ? code.length : nextTag
    result += hlEntities(code.slice(i, endPos))
    i = endPos
  }

  return result
}

function fmtHTML(html) {
  let out = '', i = 0
  for (let line of html.replace(/>\s*</g,'>\n<').split('\n')) {
    line = line.trim(); if (!line) continue
    if (line.match(/^<\/[^>]+>/)) i = Math.max(0, i - 1)
    out += '  '.repeat(i) + line + '\n'
    if (line.match(/^<[^/!][^>]*[^/]>$/) && !line.match(/^<(br|hr|img|input|meta|link)/i)) i++
  }
  return out.trim()
}

    const start = ta.selectionStart
    const end   = ta.selectionEnd
    const val   = ta.value
    ta.value = val.slice(0, start) + '  ' + val.slice(end)
    ta.selectionStart = ta.selectionEnd = start + 2
    htmlCode.value = ta.value
    codeEdited.value = true
  }
}

function htmlToMd(html) {
  return html
    .replace(/<h([1-4])[^>]*>(.*?)<\/h\1>/gi,(_, n, t) => '#'.repeat(n)+' '+t+'\n\n')
    .replace(/<strong[^>]*>(.*?)<\/strong>/gi,'**$1**').replace(/<b[^>]*>(.*?)<\/b>/gi,'**$1**')
    .replace(/<em[^>]*>(.*?)<\/em>/gi,'*$1*').replace(/<i[^>]*>(.*?)<\/i>/gi,'*$1*')
    .replace(/<a[^>]*href="([^"]*)"[^>]*>(.*?)<\/a>/gi,'[$2]($1)')
    .replace(/<img[^>]*src="([^"]*)"[^>]*/gi,'![]($1)')
    .replace(/<li[^>]*>(.*?)<\/li>/gi,'- $1\n').replace(/<blockquote[^>]*>(.*?)<\/blockquote>/gi,'> $1\n')
    .replace(/<pre[^>]*>(.*?)<\/pre>/gi,'```\n$1\n```\n').replace(/<br\s*\/?>/gi,'\n')
    .replace(/<\/p>/gi,'\n\n').replace(/<[^>]+>/g,'')
    .replace(/&nbsp;/g,' ').replace(/&lt;/g,'<').replace(/&gt;/g,'>').replace(/&amp;/g,'&')
    .replace(/\n{3,}/g,'\n\n').trim()
}
