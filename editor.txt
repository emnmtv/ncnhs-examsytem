<script setup>
import { ref, computed, onMounted, defineComponent, h } from 'vue'

// ── Load Keenicons + Inter font ───────────────────────────────────────────────
onMounted(() => {
  if (!document.getElementById('keenicons-css')) {
    const link = document.createElement('link')
    link.id = 'keenicons-css'
    link.rel = 'stylesheet'
    link.href = 'https://cdn.jsdelivr.net/npm/keenicons@2.0.0/assets/css/style.css'
    document.head.appendChild(link)
  }
  if (!document.getElementById('inter-font')) {
    const link = document.createElement('link')
    link.id = 'inter-font'
    link.rel = 'stylesheet'
    link.href = 'https://fonts.googleapis.com/css2?family=Inter:wght@300;400;500;600;700&display=swap'
    document.head.appendChild(link)
  }
  if (editorEl.value) {
    editorEl.value.innerHTML = htmlCode.value
    saveState(); updateStats()
  }
  document.addEventListener('fullscreenchange', () => {
    if (!document.fullscreenElement) isFullscreen.value = false
  })
})

// ── WBtn Component ────────────────────────────────────────────────────────────
const WBtn = defineComponent({
  props: { icon: String, title: String, active: Boolean, disabled: Boolean },
  emits: ['click'],
  slots: ['label'],
  setup(props, { emit, slots }) {
    return () => h('button', {
      title: props.title,
      disabled: props.disabled,
      class: ['wb', { 'wb--on': props.active, 'wb--off': props.disabled }],
      onMousedown: (e) => { e.preventDefault(); if (!props.disabled) emit('click') },
    }, [
      h('i', { class: `ki-outline ki-${props.icon} fs-5` }),
      slots.label ? h('span', { class: 'wb-label' }, slots.label()) : null,
    ])
  }
})

// ── Color Picker Component ────────────────────────────────────────────────────
const COLORS = [
  '#000000','#1e293b','#334155','#475569','#94a3b8','#e2e8f0',
  '#ffffff','#f8fafc','#f1f5f9','#e2e8f0','#64748b','#0f172a',

  '#14532d','#166534','#16a34a','#22c55e','#86efac','#dcfce7',
  '#1e3a8a','#1d4ed8','#2563eb','#3b82f6','#93c5fd','#dbeafe',
  '#78350f','#92400e','#d97706','#f59e0b','#fcd34d','#fef3c7',
  '#4c1d95','#6d28d9','#7c3aed','#8b5cf6','#c4b5fd','#ede9fe',
  '#164e63','#0e7490','#0891b2','#06b6d4','#67e8f9','#cffafe',
]

const ColorPick = defineComponent({
  props: { title: String },
  emits: ['pick'],
  setup(props, { emit, slots }) {
    const open = ref(false)
    const handler = (e) => { if (!e.target.closest?.('.cp-root')) open.value = false }
    onMounted(() => document.addEventListener('mousedown', handler))
    return () => h('div', { class: 'cp-root' }, [
      h('button', {
        title: props.title,
        class: ['wb', { 'wb--on': open.value }],
        onMousedown: (e) => { e.preventDefault(); open.value = !open.value }
      }, [slots.default?.()]),
      open.value && h('div', { class: 'cp-panel' }, [
        h('div', { class: 'cp-panel-title' }, props.title),
        h('div', { class: 'cp-grid' }, COLORS.map(c => h('button', {
          class: 'cp-dot',
          style: { background: c },
          onMousedown: (e) => { e.preventDefault(); emit('pick', c); open.value = false }
        }))),
        h('div', { class: 'cp-custom-row' }, [
          h('span', { class: 'cp-custom-lbl' }, 'Custom color'),
          h('input', { type: 'color', class: 'cp-custom-inp', onChange: (e) => emit('pick', e.target.value) })
        ])
      ])
    ])
  }
})

function fmtHTML(html) {
  let out = '', indent = 0
  for (let line of html.replace(/>\s*</g,'>\n<').split('\n')) {
    line = line.trim(); if (!line) continue
    if (line.match(/^<\/[^>]+>/)) indent = Math.max(0, indent - 1)
    out += '  '.repeat(indent) + line + '\n'
    if (line.match(/^<[^/!][^>]*[^/]>$/) && !line.match(/^<(br|hr|img|input|meta|link)/i)) indent++
  }
  return out.trim()
}
function showToast(msg) { toast.value = msg; setTimeout(() => toast.value = null, 2500) }
function updateStats() {
  if (!editorEl.value) return
  const text = editorEl.value.innerText || ''
  wordCount.value   = text.trim() ? text.trim().split(/\s+/).length : 0
  charCount.value   = text.length; charNoSpace.value = text.replace(/\s/g,'').length
  lineCount.value   = text.split('\n').filter(l => l.trim()).length
  paraCount.value   = editorEl.value.querySelectorAll('p,h1,h2,h3,h4,h5,h6,blockquote,pre,li').length
}

  ;['bold','italic','underline','strikeThrough','insertOrderedList','insertUnorderedList',

    try { f[cmd] = document.queryCommandState(cmd) } catch (e) { /* ignore */ }
  })
  fmt.value = f
  try { headingVal.value = (document.queryCommandValue('formatBlock').toLowerCase().replace(/[<>]/g,'')) || 'p' }
  catch (e) { /* ignore */ }
}

function handleUndo() {
  if (undoStack.value.length < 2) return
  const curr = undoStack.value.pop(); redoStack.value = [...redoStack.value, curr]
  savingFromHistory = true; if (editorEl.value) editorEl.value.innerHTML = undoStack.value[undoStack.value.length - 1] || ''
  savingFromHistory = false; updateStats(); syncCode()
}
function handleRedo() {
  if (!redoStack.value.length) return
  const next = redoStack.value.pop(); undoStack.value = [...undoStack.value, next]
  savingFromHistory = true; if (editorEl.value) editorEl.value.innerHTML = next
  savingFromHistory = false; updateStats(); syncCode()
}

}

function htmlToMd(html) {
  return html
    .replace(/<h([1-4])[^>]*>(.*?)<\/h\1>/gi, (_, n, t) => '#'.repeat(n) + ' ' + t + '\n\n')
    .replace(/<strong[^>]*>(.*?)<\/strong>/gi,'**$1**').replace(/<b[^>]*>(.*?)<\/b>/gi,'**$1**')
    .replace(/<em[^>]*>(.*?)<\/em>/gi,'*$1*').replace(/<i[^>]*>(.*?)<\/i>/gi,'*$1*')
    .replace(/<a[^>]*href="([^"]*)"[^>]*>(.*?)<\/a>/gi,'[$2]($1)')
    .replace(/<img[^>]*src="([^"]*)"[^>]*/gi,'![]($1)')
    .replace(/<li[^>]*>(.*?)<\/li>/gi,'- $1\n').replace(/<blockquote[^>]*>(.*?)<\/blockquote>/gi,'> $1\n')
    .replace(/<pre[^>]*>(.*?)<\/pre>/gi,'```\n$1\n```\n').replace(/<br\s*\/?>/gi,'\n')
    .replace(/<\/p>/gi,'\n\n').replace(/<[^>]+>/g,'').replace(/&nbsp;/g,' ')
    .replace(/&lt;/g,'<').replace(/&gt;/g,'>').replace(/&amp;/g,'&').replace(/\n{3,}/g,'\n\n').trim()
}
function exportAs(type) {
  const html = editorEl.value?.innerHTML || ''
  const map = {
    html: { content:`<!DOCTYPE html>\n<html lang="en">\n<head>\n  <meta charset="UTF-8">\n  <title>Document</title>\n  <link href="https://fonts.googleapis.com/css2?family=Inter:wght@400;600;700&display=swap" rel="stylesheet">\n  <style>body{font-family:Inter,sans-serif;max-width:820px;margin:48px auto;padding:0 32px;line-height:1.7;color:#1e293b}h1,h2,h3{color:#0f172a}pre{background:#f1f5f9;padding:16px;border-radius:8px}table{border-collapse:collapse;width:100%}td,th{border:1px solid #e2e8f0;padding:10px 14px}</style>\n</head>\n<body>\n${html}\n</body>\n</html>`, name:'document.html', mime:'text/html' },
    txt:  { content: editorEl.value?.innerText || '', name:'document.txt', mime:'text/plain' },
    md:   { content: htmlToMd(html),                  name:'document.md',  mime:'text/markdown' },
  }
  const { content, name, mime } = map[type]
  const a = Object.assign(document.createElement('a'), { href: URL.createObjectURL(new Blob([content],{type:mime})), download: name })
  a.click(); URL.revokeObjectURL(a.href); showExport.value = false; showToast(`Downloaded ${name}`)
}
function handlePrint() {
  const w = window.open('', '_blank')
  w.document.write(`<!DOCTYPE html><html><head><meta charset="UTF-8"><title>Print</title><link href="https://fonts.googleapis.com/css2?family=Inter:wght@400;600;700&display=swap" rel="stylesheet"><style>body{font-family:Inter,sans-serif;max-width:800px;margin:40px auto;padding:0 24px;line-height:1.7;color:#1e293b}@media print{body{margin:0;max-width:none}}</style></head><body>${editorEl.value?.innerHTML || ''}</body></html>`)
  w.document.close(); w.print()
}
function toggleFullscreen() {

}

  catch (e) { showToast('Copy failed') }
}

  }
}

function onInput() {
  updateStats(); updateFmts()
  if (!savingFromHistory) saveState()
  syncCode(); if (findQ.value) countFinds()
}
function onKeyDown(e) {
  if (e.ctrlKey || e.metaKey) {
    if (e.key === 'z') { e.preventDefault(); handleUndo() }
    if (e.key === 'y') { e.preventDefault(); handleRedo() }
    if (e.key === 'f') { e.preventDefault(); showFindReplace.value = true }
    if (e.key === 'p') { e.preventDefault(); handlePrint() }
  }
}
function onPaste(e) {
  if (pasteAsPlain.value) {
    e.preventDefault()
    document.execCommand('insertText', false, e.clipboardData.getData('text/plain'))
    saveState(); syncCode(); updateStats()
  }
}
